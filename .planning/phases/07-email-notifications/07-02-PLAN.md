    ```typescript
    import { cronJobs } from "convex/server";
    import { internalMutation } from "./_generated/server";
    import { internal } from "./_generated/api";

    const MS_PER_DAY = 24 * 60 * 60 * 1000;

    /**
     * Internal mutation: daily deadline scan.
     * Runs inside the Convex V8 runtime (no "use node" — no Node deps here).
     * For each due reminder, schedules sendReminderEmail via ctx.scheduler.runAfter().
     *
     * Logic (per D-08, D-09, D-10, D-21, D-23):
     * 1. Fetch all sent initial dispute letters
     * 2. For each, compute daysSinceSent relative to sentAt (NOT deadline — Pitfall 3)
     * 3. Day-25 window: [25, 26), Day-31 window: [31, 32)
     * 4. Skip if dispute_items.status !== "sent"
     * 5. Skip if any bureau_response recorded for the dispute item (D-10)
     * 6. Skip if user emailRemindersEnabled === false (D-21)
     * 7. Skip if reminder_log already has an entry for this letter+type (D-23)
     * 8. Insert reminder_log row, schedule sendReminderEmail action
     */
    export const scanDeadlines = internalMutation({
      args: {},
      handler: async (ctx) => {
        const now = Date.now();
        const trackerUrl =
          (process.env.FRONTEND_URL ?? "https://creditfix.app") + "/tracker";

        // Step 1: fetch all sent dispute letters (sentAt is defined)
        // Using by_user index is not ideal; collect all and filter in JS
        // (Convex guideline: no filter() in queries — collect by index, filter in JS)
        const allLetters = await ctx.db
          .query("dispute_letters")
          .collect();

        for (const letter of allLetters) {
          try {
            // Skip if not sent (sentAt absent)
            if (letter.sentAt === undefined) continue;

            // Skip demand/escalation letters — only trigger on initial letters (D-08, D-09)
            if (
              letter.letterType === "demand" ||
              letter.letterType === "escalation"
            ) continue;

            // Compute day offset from sentAt (NOT from deadline — Pitfall 3)
            const daysSinceSent = (now - letter.sentAt) / MS_PER_DAY;

            // Determine which reminder type applies
            let reminderType: "day25" | "day31" | null = null;
            if (daysSinceSent >= 25 && daysSinceSent < 26) {
              reminderType = "day25";
            } else if (daysSinceSent >= 31 && daysSinceSent < 32) {
              reminderType = "day31";
            }
            if (!reminderType) continue;

            // Step 4: fast pre-filter — check dispute_items.status
            const disputeItem = await ctx.db.get(letter.disputeItemId);
            if (!disputeItem || disputeItem.status !== "sent") continue;

            // Step 5: check for recorded bureau response (D-10, Pitfall 4)
            const existingResponse = await ctx.db
              .query("bureau_responses")
              .withIndex("by_dispute_item", (q) =>
                q.eq("disputeItemId", letter.disputeItemId)
              )
              .first();
            if (existingResponse) continue;

            // Step 6: load user preferences (D-21)
            const user = await ctx.runQuery(internal.users.getEmailPrefs, {
              userId: letter.userId,
            });
            if (!user) continue;

            // Respect opt-out (undefined treated as true — enabled by default per D-21)
            if (user.emailRemindersEnabled === false) continue;

            // Resolve destination email (Pitfall 5)
            const toEmail = user.reminderEmail || user.email;
            if (!toEmail) continue;  // no email address on record — skip silently

            // Step 7: de-duplication check (D-23)
            const existingLog = await ctx.db
              .query("reminder_log")
              .withIndex("by_letter_and_type", (q) =>
                q.eq("letterId", letter._id).eq("reminderType", reminderType!)
              )
              .first();
            if (existingLog) continue; // already sent for this letter+type

            // Step 8a: insert reminder_log row (before scheduling — prevents double-send if scheduler fails)
            await ctx.db.insert("reminder_log", {
              letterId:     letter._id,
              userId:       letter.userId,
              reminderType,
              sentAt:       now,
            });

            // Step 8b: schedule the email send action (D-07)
            await ctx.scheduler.runAfter(
              0,
              internal.notifications.sendReminderEmail,
              {
                toEmail,
                creditorName: disputeItem.creditorName,
                bureau:       letter.bureau,
                sentAt:       letter.sentAt,
                reminderType,
                trackerUrl,
              },
            );
          } catch (err) {
            // Per-letter catch — log and continue; cron must not halt on one failure
            console.error("scanDeadlines: error processing letter", letter._id, err);
          }
        }
      },
    });

    // Register the daily cron job.
    // CRITICAL: use crons.cron() — crons.daily() is forbidden by Convex AI guidelines (Pitfall 1).
    const crons = cronJobs();

    crons.cron(
      "daily-deadline-scan",
      "0 8 * * *",              // 8 AM UTC every day
      internal.crons.scanDeadlines,
      {},
    );

    export default crons;
    ```
